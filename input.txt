#include <iostream>
#include <string>

/* Variabile globale */
int globalVar = 10; // Variabilă globală corectă
string globalStr = 20; // Inițializare invalidă
float pi = 3.14;

/* Funcție iterativă */
void iterativeFunction(int n) {
    int localVar = 5; // Variabilă locală corectă
    string localStr = "Hello, world!"; // Inițializare corectă
    int localVar = 15; // Redeclarație invalidă
    float pi = 6;

    for (int i = 0; i < n; i++) {
        localVar += i;
        // Comentariu care ar trebui eliminat
        if (localVar > 50) {
            break;
        }
    }
}

/* Funcție recursivă */
int recursiveFunction(int n) {
    if (n <= 1) {
        return n;
    }
    return n + recursiveFunction(n - 1); // Apel recursiv valid
}

int main() {
    int x = 5; // Inițializare validă
    string s = 123; // Inițializare invalidă
    float y = 1.234; // Inițializare validă

        x += i; // Operare validă

            // Ieșire din buclă dacă x > 50
            break;
        }
    }
